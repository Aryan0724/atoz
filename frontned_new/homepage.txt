    <script>
        // --- 1. SETUP ---
        gsap.registerPlugin(ScrollTrigger, ScrollToPlugin, Draggable);
        lucide.createIcons();

        // Lenis Smooth Scroll
        const lenis = new Lenis({
            duration: 1.5,
            easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
            direction: 'vertical',
            smooth: true,
        });

        function raf(time) {
            lenis.raf(time);
            requestAnimationFrame(raf);
        }
        requestAnimationFrame(raf);

        // --- DOM LOADED WRAPPER ---
        document.addEventListener('DOMContentLoaded', () => {
            
            // --- LOADER ---
            const tl = gsap.timeline();
            tl.to(".loader-overlay", { yPercent: -100, duration: 1.5, ease: "power4.inOut", delay: 0.5 })
              .to(".hero-text", { y: 0, duration: 1.5, stagger: 0.1, ease: "power4.out" }, "-=0.8")
              .to(".hero-reveal", { y: 0, opacity: 1, duration: 1.2, stagger: 0.1, ease: "power3.out" }, "-=1.2")
              .to(".hero-img-cover", { scaleY: 0, duration: 1.5, ease: "power4.inOut" }, "-=1.5")
              .from(".hero-img", { scale: 1.3, duration: 1.8, ease: "power2.out" }, "-=1.2")
              .to(".hero-badge", { opacity: 1, y: 0, duration: 1, ease: "back.out(1.7)" }, "-=0.5");

            // --- MANIFESTO ANIMATIONS ---
            if (typeof SplitType !== 'undefined') {
                const type = new SplitType('#manifesto-text', { types: 'words' });
                const words = type.words;
                
                gsap.set(words, { color: 'rgba(255,255,255,0.1)' });

                gsap.to(words, {
                    color: '#F9F9F7',
                    textShadow: "0 0 10px rgba(255,255,255,0.3)",
                    stagger: 0.1,
                    scrollTrigger: {
                        trigger: "#manifesto",
                        start: "top 60%",
                        end: "center center",
                        scrub: 1.5
                    }
                });

                gsap.utils.toArray('.reveal-stat').forEach((stat, i) => {
                    gsap.to(stat, {
                        opacity: 1,
                        y: 0,
                        duration: 1,
                        delay: i * 0.2,
                        scrollTrigger: {
                            trigger: "#manifesto",
                            start: "center 70%"
                        }
                    });
                });
            }

            // --- GLOBAL TEXT REVEAL (TITLES & PARAGRAPHS) ---
            if (typeof SplitType !== 'undefined') {
                // Titles
                const titles = document.querySelectorAll('.reveal-title');
                titles.forEach(title => {
                    const splitTitle = new SplitType(title, { types: 'lines, words' });
                    gsap.from(splitTitle.words, {
                        y: 50,
                        opacity: 0,
                        duration: 1.2,
                        stagger: 0.05,
                        ease: "power3.out",
                        scrollTrigger: {
                            trigger: title,
                            start: "top 85%"
                        }
                    });
                });

                // Paragraphs
                const texts = document.querySelectorAll('.reveal-text');
                texts.forEach(text => {
                    const splitText = new SplitType(text, { types: 'lines' });
                    gsap.from(splitText.lines, {
                        y: 30,
                        opacity: 0,
                        duration: 1,
                        stagger: 0.1,
                        ease: "power3.out",
                        scrollTrigger: {
                            trigger: text,
                            start: "top 90%"
                        }
                    });
                });
            }

            // --- COLLECTION REVEAL (STAGGERED) ---
            gsap.utils.toArray('.reveal-item').forEach((item, i) => {
                gsap.from(item, {
                    y: 100,
                    opacity: 0,
                    duration: 1.2,
                    ease: "power3.out",
                    scrollTrigger: {
                        trigger: item,
                        start: "top 90%"
                    }
                });
            });

            // --- IMPACT STATS ANIMATION ---
            const counters = document.querySelectorAll(".counter");
            const bars = document.querySelectorAll(".stat-bar");

            ScrollTrigger.create({
                trigger: "#impact",
                start: "top 60%",
                onEnter: () => {
                    counters.forEach(counter => {
                        let target = +counter.getAttribute('data-target');
                        gsap.to(counter, {
                            innerText: target,
                            duration: 2,
                            snap: { innerText: 1 },
                            ease: "power2.out"
                        });
                    });
                    bars.forEach(bar => {
                        gsap.to(bar, { height: '100%', duration: 1.5, ease: "power4.out", stagger: 0.2 });
                    });
                }
            });

            // --- EXTREME SMOOTH DRAGGABLE COMPARISON ---
            const container = document.getElementById("compare-container");
            const wrapper = document.getElementById("before-wrapper");
            const handle = document.getElementById("compare-handle");

            if (container && wrapper && handle) {
                // Initial set - CENTERED HANDLE
                gsap.set(handle, { x: container.offsetWidth * 0.5 });
                gsap.set(wrapper, { width: container.offsetWidth * 0.5 });

                Draggable.create(handle, {
                    type: "x",
                    bounds: container,
                    inertia: true,
                    onDrag: function() {
                        let x = this.x; // Gets current position relative to bounds
                        // Draggable x includes transform, we need offset from left
                        // Since we set handle inside container with absolute positioning
                        // Draggable modifies transform translateX
                        
                        // Actually, easier approach: center handle at 50%, then draggable moves it relative to that.
                        // Let's reset: use a proxy or just map the handle's center to wrapper width
                        
                        let percent = (this.x / container.offsetWidth) * 100;
                        
                        // Constrain percentage visually if needed, but bounds handle it mostly
                        if(percent < 0) percent = 0;
                        if(percent > 100) percent = 100;
                        
                        gsap.set(wrapper, { width: this.x });
                    },
                    onPress: function() {
                        gsap.to(handle, { scale: 1.2, duration: 0.2 });
                    },
                    onRelease: function() {
                        gsap.to(handle, { scale: 1, duration: 0.2 });
                    }
                });
            }

            // --- CALCULATOR LOGIC (FIXED) ---
            const productBtns = document.querySelectorAll('.product-btn');
            const qtySlider = document.getElementById('qty-slider');
            const qtyDisplay = document.getElementById('qty-display');
            const finalPrice = document.getElementById('final-price');
            const unitCostDisplay = document.getElementById('unit-cost');
            const calcOpts = document.querySelectorAll('.calc-opt');
            
            if(qtySlider && finalPrice) {
                let currentBasePrice = 2; // Default for first item

                // 1. Setup Product Buttons
                productBtns.forEach(btn => {
                    btn.addEventListener('click', () => {
                        // Styles
                        productBtns.forEach(b => {
                            b.classList.remove('active', 'bg-white', 'text-brand-darkBlue', 'border-transparent');
                            b.classList.add('border-white/20');
                        });
                        btn.classList.add('active', 'bg-white', 'text-brand-darkBlue', 'border-transparent');
                        btn.classList.remove('border-white/20');
                        
                        // Logic
                        currentBasePrice = parseFloat(btn.getAttribute('data-price'));
                        calculateTotal();
                    });
                });

                // 2. Setup Active State for First Button
                if(productBtns.length > 0) {
                    const firstBtn = productBtns[0];
                    firstBtn.classList.add('bg-white', 'text-brand-darkBlue', 'border-transparent');
                    firstBtn.classList.remove('border-white/20');
                }

                // 3. Calculation Function
                function calculateTotal() {
                    let qty = parseInt(qtySlider.value);
                    let unitCost = currentBasePrice;
                    
                    // Add options
                    calcOpts.forEach(opt => {
                        if(opt.checked) {
                            // Extract cost from ID map or simple logic
                            // Map: foil=5, uv=3, texture=4, design=fixed 2500
                            if(opt.id === 'opt-foil') unitCost += 5;
                            if(opt.id === 'opt-uv') unitCost += 3;
                            if(opt.id === 'opt-texture') unitCost += 4;
                        }
                    });

                    let total = qty * unitCost;

                    // Fixed Design Cost
                    const designOpt = document.getElementById('opt-design');
                    if(designOpt && designOpt.checked) {
                        total += 2500;
                    }

                    // DOM Updates
                    qtyDisplay.innerText = qty;
                    unitCostDisplay.innerText = "₹" + unitCost.toFixed(2);
                    
                    // Simple animation
                    gsap.to(finalPrice, {
                        duration: 0.5,
                        scrambleText: {
                            text: "₹" + Math.floor(total).toLocaleString(),
                            chars: "0123456789",
                            revealDelay: 0.1,
                            speed: 0.3
                        }
                    });
                    // Fallback if scrambleText not available (it's a paid plugin usually, standard text update below)
                    if(!gsap.plugins.scrambleText) {
                         finalPrice.innerText = "₹" + Math.floor(total).toLocaleString();
                    }
                }

                // 4. Listeners
                qtySlider.addEventListener('input', calculateTotal);
                calcOpts.forEach(opt => opt.addEventListener('change', calculateTotal));
                
                // Initial calc
                calculateTotal();
            }

            // --- MAGNETIC CURSOR ---
            const cursorDot = document.querySelector('.cursor-dot');
            const cursorCircle = document.querySelector('.cursor-circle');
            
            window.addEventListener('mousemove', (e) => {
                gsap.to(cursorDot, { x: e.clientX, y: e.clientY, duration: 0.1 });
                gsap.to(cursorCircle, { x: e.clientX, y: e.clientY, duration: 0.3 });
            });
            
            document.querySelectorAll('.magnetic-target').forEach(el => {
                el.addEventListener('mouseenter', () => {
                    document.body.classList.add('hovering');
                    gsap.to(cursorCircle, { scale: 1.5 });
                });
                el.addEventListener('mouseleave', () => {
                    document.body.classList.remove('hovering');
                    gsap.to(cursorCircle, { scale: 1 });
                });
            });

            // --- NAVBAR SCROLL EFFECT ---
            const navbar = document.getElementById('navbar');
            window.addEventListener('scroll', () => {
                if (window.scrollY > 50) {
                    navbar.classList.add('scrolled');
                } else {
                    navbar.classList.remove('scrolled');
                }
            });

        }); // End DOMContentLoaded

        // FAQ Toggle
        function toggleAccordion(element) {
            const content = element.querySelector('.accordion-content');
            const isActive = element.classList.contains('active');
            
            // Close all
            document.querySelectorAll('.accordion-item').forEach(item => {
                item.classList.remove('active');
            });

            if (!isActive) {
                element.classList.add('active');
            }
        }

    </script>
