    <script>
        gsap.registerPlugin(ScrollTrigger, Draggable);
        lucide.createIcons();

        const lenis = new Lenis({ duration: 1.5, easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)), direction: 'vertical', smooth: true });
        function raf(time) { lenis.raf(time); requestAnimationFrame(raf); }
        requestAnimationFrame(raf);

        document.addEventListener('DOMContentLoaded', () => {
            // Loader
            const tl = gsap.timeline();
            tl.to(".loader-overlay", { yPercent: -100, duration: 1.5, ease: "power4.inOut", delay: 0.5 })
              .to(".hero-text", { y: 0, duration: 1.5, stagger: 0.1, ease: "power4.out" }, "-=0.8")
              .to(".hero-reveal", { y: 0, opacity: 1, duration: 1.2, stagger: 0.1, ease: "power3.out" }, "-=1.2");

            // Text Reveal
            if (typeof SplitType !== 'undefined') {
                const titles = document.querySelectorAll('.reveal-title');
                titles.forEach(title => {
                    const splitTitle = new SplitType(title, { types: 'lines, words' });
                    gsap.from(splitTitle.words, { y: 50, opacity: 0, duration: 1.2, stagger: 0.05, ease: "power3.out", scrollTrigger: { trigger: title, start: "top 85%" } });
                });
            }

            // Philosophy Animation (Clean Fade Out Up)
            gsap.utils.toArray('.philosophy-card').forEach((item, i) => {
                ScrollTrigger.create({
                    trigger: item,
                    start: "top center",
                    end: "bottom top", 
                    onUpdate: self => {
                        // Fade as it moves UP (progress > 0)
                        if(self.progress > 0.2) {
                            gsap.to(item, { opacity: 1 - self.progress, y: -50 * self.progress, duration: 0.5 });
                        } else {
                            gsap.to(item, { opacity: 1, y: 0, duration: 0.5 });
                        }
                    }
                });
            });

            // Timeline Animation (Center Line & Active State)
            gsap.to(".timeline-line-fill", {
                height: "100%",
                ease: "none",
                scrollTrigger: {
                    trigger: ".timeline-container",
                    start: "top center",
                    end: "bottom center",
                    scrub: 0.5
                }
            });

            gsap.utils.toArray('.timeline-row').forEach(row => {
                ScrollTrigger.create({
                    trigger: row,
                    start: "top 60%",
                    end: "bottom 40%",
                    onToggle: self => {
                        if (self.isActive) row.classList.add('active');
                        else row.classList.remove('active');
                    }
                });
                // Vertical Reveal
                gsap.from(row.querySelector('.timeline-text'), { y: 50, opacity: 0, duration: 1, scrollTrigger: { trigger: row, start: "top 80%" } });
                gsap.from(row.querySelector('.timeline-img'), { y: 80, opacity: 0, duration: 1.2, scrollTrigger: { trigger: row, start: "top 80%" } });
            });

            // Process Animation (Circuit Line)
            gsap.to(".process-line-fill", {
                height: "100%", ease: "none",
                scrollTrigger: { trigger: ".process-line-center", start: "top center", end: "bottom center", scrub: 0.5 }
            });
            gsap.utils.toArray('.process-card').forEach(card => {
                gsap.from(card, {
                    y: 50, opacity: 0, duration: 1, ease: "power3.out",
                    scrollTrigger: { trigger: card, start: "top 85%" }
                });
            });

            // Parallax
            gsap.to(".parallax-img", { yPercent: 20, ease: "none", scrollTrigger: { trigger: ".parallax-img", start: "top bottom", end: "bottom top", scrub: true } });

            // Team Hover (Fixed Image)
            const teamItems = document.querySelectorAll('.team-item');
            const teamImages = document.querySelectorAll('.team-image-fixed');

            teamItems.forEach(item => {
                item.addEventListener('mouseenter', () => {
                    const index = item.getAttribute('data-index');
                    teamImages.forEach(img => img.classList.remove('active'));
                    const target = document.getElementById(`team-img-${index}`);
                    if(target) target.classList.add('active');
                });
            });

            // Map Lines Reveal
            gsap.to(".map-connection", {
                width: "200px", opacity: 0.5, duration: 2, ease: "power2.out",
                scrollTrigger: { trigger: ".map-connection", start: "top 80%" }
            });

            // Navbar Scroll
            const navbar = document.getElementById('navbar');
            window.addEventListener('scroll', () => {
                if (window.scrollY > 50) navbar.classList.add('scrolled');
                else navbar.classList.remove('scrolled');
            });

            // Cursor
            const cursorDot = document.querySelector('.cursor-dot');
            const cursorCircle = document.querySelector('.cursor-circle');
            window.addEventListener('mousemove', (e) => {
                gsap.to(cursorDot, { x: e.clientX, y: e.clientY, duration: 0.1 });
                gsap.to(cursorCircle, { x: e.clientX, y: e.clientY, duration: 0.3 });
            });
            document.querySelectorAll('.magnetic-target').forEach(el => {
                el.addEventListener('mouseenter', () => {
                    document.body.classList.add('hovering');
                    gsap.to(cursorCircle, { scale: 1.5 });
                });
                el.addEventListener('mouseleave', () => {
                    document.body.classList.remove('hovering');
                    gsap.to(cursorCircle, { scale: 1 });
                });
            });
        });
